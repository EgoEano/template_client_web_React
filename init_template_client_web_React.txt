# OPTIONAL How to init manually

# Initialize project
	npm init -y
	
# Dev packs
	npm install --save-dev dotenv cross-env
	
# Installing Webpack (as client builder)
	npm install --save-dev webpack webpack-cli webpack-dev-server html-webpack-plugin webpack-obfuscator mini-css-extract-plugin css-minimizer-webpack-plugin

# Create webpack.config.ts

# For TypeScript
	npm install --save-dev typescript ts-loader ts-node @types/node

# Configure TypeScript
	npx tsc --init

# Change type in package.json
	"type": "module"

# Main packs (react logic)
	npm install react react-dom react-router-dom react-native react-native-web
